def print_madlib
        @word_array.each do |h|
                h.each do |key, value|
                        @word = "@@@" << key
                        #puts "replace #{@word} with #{value}"
                        @story.sub!(/#{Regexp.quote(@word)}/, value)
                        #@story.sub!(/@@@\w*\b/, value)
                end
        end
        puts @story
end
